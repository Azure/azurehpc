## Cluster Configuration File ##

[cluster pbs_azhpc_beegfs]
FormLayout = selectionpanel
Category = Schedulers

Autoscale = $Autoscale

    [[node defaults]]
    UsePublicNetwork = $UsePublicNetwork
    Credentials = $Credentials    
    ImageName = $ImageName
    SubnetId = $SubnetId
    Region = $Region
    KeyPairLocation = ~/.ssh/cyclecloud.pem
    
        [[[configuration]]]
        cyclecloud.mounts.sched.disabled = true
        cyclecloud.mounts.shared.disabled = true
        cshared.server.legacy_links_disabled = true
        cuser.base_home_dir = /shared/home
        beegfs.mgmt_host = $BeeGFSMgmtHost

        [[[cluster-init cyclecloud/pbspro:default]]]
        [[[cluster-init azhpc:default:1.0.0]]]
        Optional = false

    [[node master]]
        MachineType = $MasterMachineType
        IsReturnProxy = $ReturnProxy
        AdditionalClusterInitSpecs = $MasterClusterInitSpecs

        [[[volume disk0]]]
        Size = $DiskSize
        Persistent = true
        SSD = True
        Mount = home

        [[[volume disk1]]]
        Size = $DiskSize
        Persistent = true
        SSD = True
        Mount = home

        [[[configuration cyclecloud.mounts.home]]]
        mountpoint = /shared/home
        fs_type = xfs
        raid_level = 0
    
        [[[configuration]]]

        [[[configuration cyclecloud.exports.home]]]
        type = nfs
        export_path = /shared/home
        options = no_root_squash

        [[[cluster-init cyclecloud/pbspro:master]]]
        [[[cluster-init azhpc:beegfs-client:1.0.0]]]

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $UsePublicNetwork

        [[[input-endpoint ganglia]]]
        PrivatePort = 8652
        PublicPort = 8652

    [[nodearray login]]
    InitialCount = $NumberLoginNodes
    MachineType = $MasterMachineType
        
	[[[cluster-init cyclecloud/pbspro:login]]]
    [[[configuration]]]
    autoscale.disabled = true

    [[nodearray execute]]
        MachineType = $ExecuteMachineType
        MaxCoreCount = $MaxExecuteCoreCount

        Interruptible = $UseLowPrio
        AdditionalClusterInitSpecs = $ExecuteClusterInitSpecs

        [[[configuration cyclecloud.mounts.home]]]
        type = nfs
        mountpoint = /shared/home
        export_path = /shared/home

        [[[configuration]]]
        autoscale.disabled = ${Autoscale == False}
        
        [[[cluster-init cyclecloud/pbspro:execute]]]
        [[[cluster-init azhpc:beegfs-client:1.0.0]]]

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $ExecuteNodesPublic

    [[parameters Auto-Scaling]]
